% --- Kotak Pendalaman & Sintesis (tcolorbox, kontras via tone) --------------
\usepackage[most]{tcolorbox}

% Pendalaman: sidebar abu muda + palang kiri gelap. Kepala dua baris:
% baris-1 banner "PENDALAMAN", baris-2 judul topik, dipisah garis rambut.
\newtcolorbox{pendalaman}[1]{
  enhanced, breakable, sharp corners,
  colback=panelbg, colframe=panelbg, boxrule=0pt,
  borderline west={3pt}{0pt}{panelbar},
  left=13pt, right=11pt, top=9pt, bottom=10pt,
  before upper={%
    {\sffamily\bfseries\footnotesize\textcolor{panelbar}{\MakeUppercase{Pendalaman}}}\par
    \vspace{1.5pt}%
    {\sffamily\bfseries #1}\par
    \vspace{3pt}{\color{panelbar!55}\hrule height 0.4pt}\vspace{6pt}%
    \setlength{\parindent}{0pt}\setlength{\parskip}{0.4\baselineskip}%
  },
}

% Sintesis bab: kotak penutup, lebih tegas (garis atas tebal + latar tipis).
% Antar-paragraf diberi jarak agar terbaca sebagai prosa, bukan blok padat.
\newtcolorbox{sintesis}{
  enhanced, breakable, sharp corners,
  colback=black!4, colframe=panelbar,
  boxrule=0pt, toprule=1.6pt,
  left=13pt, right=13pt, top=11pt, bottom=12pt,
  coltitle=black, fonttitle=\sffamily\bfseries\large,
  title={Sintesis Bab~\thechapter},
  before upper={\setlength{\parindent}{0pt}\setlength{\parskip}{0.5\baselineskip}},
}

    {\sffamily\bfseries\footnotesize\color{accent}\MakeUppercase{Situs \& Notebook Pendamping}}\par
    \vspace{3pt}{\color{rulegray!55}\hrule height 0.4pt}\vspace{9pt}%
    \setlength{\parindent}{0pt}%
  },
}

% Bacaan Lanjutan: BUKAN kotak (tak berisian, tak berbingkai penuh) — sengaja
% dibedakan dari Pendalaman (panel berisian) & Sintesis (pita berjudul).
% Tampil sebagai seksi daftar-baca: kepala kecil + garis rambut, butir
% berpenanda panah dengan jarak lapang, ditutup garis rambut kedua.
\usepackage{enumitem}
\newenvironment{bacaan}{%
  \par\addvspace{11pt}%
  \noindent{\color{rulegray}\rule{\linewidth}{0.6pt}}\par\vspace{5pt}%
  \noindent{\sffamily\bfseries\footnotesize\color{accent}\MakeUppercase{Bacaan Lanjutan}}%
  \hfill{\sffamily\scriptsize\itshape\color{black!55}di luar buku ini}\par
  \vspace{5pt}%
  \begingroup
  \footnotesize\linespread{0.95}\selectfont\raggedright\setlength{\parindent}{0pt}%
  \setlist[itemize,1]{%
    label=\textcolor{panelbar}{$\rightarrow$},
    leftmargin=1.3em, labelsep=0.5em, itemindent=0pt,
    itemsep=4pt, topsep=0pt, parsep=1pt}%
}{%
  \endgroup
  \par\vspace{3pt}\noindent{\color{rulegray}\rule{\linewidth}{0.6pt}}\par\addvspace{9pt}%
}

% Pembatas huruf abjad pada Glosarium (Lampiran) — kepala kecil beraksen +
% garis rambut; tak bernomor, tak masuk TOC.

  \par\addvspace{9pt}%
  \noindent{\sffamily\bfseries\large\color{panelbar}#1}\par
  \vspace{1pt}{\color{rulegray}\hrule height 0.4pt}\vspace{6pt}%
  \setlength{\parindent}{0pt}}

% Kepala "Rujukan" (daftar rujukan per bab) — tak bernomor, tak masuk TOC.

  \vspace{2pt}{\color{rulegray}\hrule height 0.4pt}\vspace{7pt}%
  \markright{Rujukan}}
